115 if(is_tty && (curr%period == 0)) {

121 int dn = curr - n_old;
122 n_old = curr;
123 t_old = now;
124 if (deq_n.size() >= smoothing) deq_n.erase(deq_n.begin());
125 if (deq_t.size() >= smoothing) deq_t.erase(deq_t.begin());
126 deq_t.push_back(dt);
127 deq_n.push_back(dn);
128
129 double avgrate = 0.;
130 if (use_ema) {
131 avgrate = deq_n[0] / deq_t[0];
132 for (unsigned int i = 1; i < deq_t.size(); i++) {
133 double r = 1.0*deq_n[i]/deq_t[i];
134 avgrate = alpha_ema*r + (1.0-alpha_ema)*avgrate;
135 }
136 } else {
137 double dtsum = std::accumulate(deq_t.begin(),deq_t.end(),0.);
138 int dnsum = std::accumulate(deq_n.begin(),deq_n.end(),0.);
139 avgrate = dnsum/dtsum;
140 }

143 // and slowing down the loop, shoot for ~25Hz and smooth over 3 seconds
144 if (nupdates > 10) {

146 smoothing = 25*3;
